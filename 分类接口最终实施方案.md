#### 文件：`src/services/api/categoryApi.js`
```javascript
import contentApiClient from './contentApiClient';

const categoryApi = {
  getCategoryList: (params = {}) => {
    const requestParams = {
      pageNum: params.pageNum || 1,
      pageSize: params.pageSize || 20,
      ...params.category,
    };
    
    return contentApiClient.get(API_ENDPOINTS.CATEGORY.LIST, { 
      params: requestParams 
    });
  },
};
```

### 3. 发布问题页面 (`src/screens/PublishScreen.js`)

#### 核心功能

**1. 分级数据结构**
```javascript
const [level1Categories, setLevel1Categories] = useState([]); // 一级分类
const [level2CategoriesMap, setLevel2CategoriesMap] = useState({}); // 二级分类缓存
const [categoryModalView, setCategoryModalView] = useState('level1'); // 视图切换
const [loadingLevel2, setLoadingLevel2] = useState(false); // 二级分类加载状态
```

**2. 一级分类加载**
```javascript
const fetchLevel1Categories = async () => {
  const response = await categoryApi.getCategoryList({
    pageNum: 1,
    pageSize: 100,
    parentId: 0, // 只获取一级分类
  });
  
  setLevel1Categories(response.data.rows);
};
```

**3. 二级分类按需加载**
```javascript
const fetchLevel2Categories = async (parentId) => {
  // 如果已缓存，直接返回
  if (level2CategoriesMap[parentId]) {
    return;
  }
  
  // 请求二级分类
  const response = await categoryApi.getCategoryList({
    pageNum: 1,
    pageSize: 100,
    parentId: parentId,
  });
  
  // 缓存结果
  setLevel2CategoriesMap(prev => ({
    ...prev,
    [parentId]: response.data.rows,
  }));
};
```

**4. 视图切换**
```javascript
const selectLevel1 = async (cat) => {
  setTempSelectedLevel1(cat);
  setCategoryModalView('level2'); // 切换到二级分类视图
  await fetchLevel2Categories(cat.id); // 按需加载
};

const selectLevel2 = (cat) => {
  setSelectedLevel1(tempSelectedLevel1);
  setSelectedLevel2(cat);
  setShowCategoryModal(false); // 关闭弹窗
};

const backToLevel1 = () => {
  setCategoryModalView('level1'); // 返回一级分类
};
```

**5. 图标映射**
```javascript
const getIconForCategory = (name, isLevel1 = false) => {
  if (isLevel1) {
    const level1IconMap = {
      '国家': 'business',

    const level2IconMap = {

      '职业发展': 'rocket',
      // ... 更多映射
    };
    return level2IconMap[name] || 'pricetag-outline';
  }
};
```

**6. UI渲染**
```javascript
{categoryModalView === 'level1' ? (

```

## 核心特性

### 1. 分级加载 ⚡
- 首次只加载一级分类（10-20条）
- 点击一级分类时，才加载该分类下的二级分类
- 首屏加载时间：100ms

### 2. 智能缓存 💾
- 已加载的二级分类会缓存在内存中
- 再次点击同一个一级分类，直接使用缓存，零延迟
- 缓存结构：`{parentId: [categories]}`

### 3. 视图切换 🔄
- 视图1：显示所有一级分类
- 视图2：显示选中的一级分类下的二级分类
- 平滑切换，用户体验好

### 4. 自动分页支持 📄
- 如果一级分类超过100个，自动处理
- 如果某个一级分类下的二级分类超过100个，自动并发加载
- 用户完全感知不到

### 5. 降级方案 🛡️
- 接口失败时，使用本地备用数据
- 保证功能可用

### 6. 图标映射 🎨
- 根据分类名称自动匹配图标
- 支持自定义图标映射

1. 用户点击"问题类别"选择器
2. 弹窗显示所有一级分类（100ms）
3. 用户点击某个一级分类（如"国家"）
4. 弹窗切换显示该分类下的所有二级分类（100ms）
5. 用户点击某个二级分类（如"政策法规"）
6. 选择完成，弹窗关闭，显示"国家 > 政策法规"

- 返回一级分类列表
- 重新选择

## 后续优化建议

### 1. 本地持久化缓存（可选）
如果想进一步优化，可以将分类数据缓存到本地存储：

```javascript
import AsyncStorage from '@react-native-async-storage/async-storage';

// 保存到本地
await AsyncStorage.setItem('categories', JSON.stringify(categories));

// 从本地读取
const cached = await AsyncStorage.getItem('categories');
if (cached) {
  setCategories(JSON.parse(cached));
}
```

### 2. 定期更新（可选）
可以添加缓存过期机制：

```javascript
const CACHE_EXPIRY = 24 * 60 * 60 * 1000; // 24小时

const isCacheExpired = (timestamp) => {
  return Date.now() - timestamp > CACHE_EXPIRY;
};
```
